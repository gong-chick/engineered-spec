`src/store/modules/order/index.ts`：

```typescript
import { ref, computed } from 'vue'
import { defineStore } from 'pinia'
import type { OrderItem } from './type'
import { getOrderListApi, deleteOrderApi } from '@/api/order'

export const useOrderStore = defineStore('order', () => {
  const orderList = ref<OrderItem[]>([])
  const currentOrder = ref<OrderItem | null>(null)
  const isLoading = ref(false)

  const orderCount = computed(() => orderList.value.length)

  async function fetchOrderList() {
    isLoading.value = true
    try {
      const result = await getOrderListApi({ page: 1, pageSize: 20 })
      orderList.value = result.list
    } finally {
      isLoading.value = false
    }
  }

  async function removeOrder(id: string) {
    await deleteOrderApi(id)
    orderList.value = orderList.value.filter(item => item.id !== id)
  }

  return { orderList, currentOrder, isLoading, orderCount, fetchOrderList, removeOrder }
})
```
